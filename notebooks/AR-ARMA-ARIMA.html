class VarMap():
    
    def __init__(self, a):
        self.w = a
        self.i = 0
        self.data = np.zeros(a)
        
    def __call__(self, x):
        self.data[self.i % self.w] = x
        
        if self.w == 1:
            return abs(x)
        
        self.i = self.i + 1
        return self.data[0:min(self.i, self.w)].std()
