Quadrature Gaussienne

on va choisir les poids et les noeuds en mme temps

\[ \int_a^b f(x)dx \approx \sum_{j = 1}^n w_j f(x_j) \] call : \(1 \times n \times f\)

L'approximation est exacte si la fonction est un polynme de degr \(2n + 1\)

forme gnrale:

Orthogonaux \[ \int_a^b \varphi_k (x) \varphi_l(x) w(x) dx = 0 \ si\ k \not= l\]

Orthonormaux \[ \int_a^b \varphi_k (x) \varphi_l(x) w(x) dx =\begin{align*} 0 \ si\ k \not= l \\ 1 \ si\ k = l \end{align*} \]

Weights are given by \[ w_i = - \frac{q_{n + 1} / q_n}{\varphi_n' \varphi_{n + 1}(x_i)} \]

def gauss_legendre(f, a, b, n):
    if n == 1:
        t = [0.0]
        w = [2.0]
    elif n == 2:
        t = [-1.0/sqrt(3.0), +1.0/sqrt(3.0)]
        w = [1.0, 1.0]
    else:
        t = [-sqrt(3.0/5.0), 0.0, +sqrt(3.0/5.0)]
        w = [5.0/9.0, 8.0/9.0, 5.0/9.0]
    
    ret = 0
    for i in range(0, min(n, 3)):
        ret = ret + f((b - a) * t[i]/2.0 + (b + a)/2.0) * w[i]
        
    return  ret * (b - a) / 2.0

def gauss_chebyshev_first(f, a, b, n):
    t = lambda x: cos((2.0 * x - 1.0) * pi / (2.0 * n))
    w = lambda x: pi / n
    ret = 0
    
    for i in range(1, n + 1):
        y = (b - a) * t(i)/2.0 + (b + a)/2.0
        
        ret = ret + f(y) * sqrt(1.0 - t(i) ** 2.0)
        
    return ret * (b - a) * pi / (2.0 * n)

def gauss_chebyshev_second(f, a, b, n):
    t = lambda x: cos(x * pi / (n + 1.0))
    w = lambda x:(sin(x * pi / (n + 1.0)) ** 2) * pi / (n + 1)
    ret = 0
    
    for i in range(1, n + 1):
        
        ret = ret + f((b - a) * t(i)/2.0 + (b + a)/2.0) * w(i) / sqrt(1.0 - t(i) ** 2.0)
        
    return (b - a) * ret / 2.0

def monte_carlo_inte(f, a, b, n):
    ret = 0
    for i in range(0, n):
        r = np.random.uniform(a, b)
        ret = ret + f(r)
        
    return (b - a) * ret/n

print 'Monte Carlo : ' + str(monte_carlo_inte(f, a, b, 10000))
