# define helper function to compute variance and returns
def portfolio_variance(weight):
    w = weight[0:n]    # this step make sure w has the right size to handle
                            # the case were weight include the risk free weight
    return np.transpose(w).dot(covariance).dot(w)

def portfolio_return(weight, rf=0.0):
    """ Return the expected portfolio return given the weight"""
    w = weight[0:n]     # this step make sure w has the right size to handle
                        # the case were weight include the risk free weight
    if rf != 0:
        return np.transpose(w).dot(expected_returns) + rf * weight[n]
    else:
        return np.transpose(w).dot(expected_returns)

Equation System

\[ \begin{eqnarray*} L(W, \lambda_{1}) & = & \frac{1}{2}W_{1 \times n}^{T} \Sigma_{n \times n} W_{n \times 1} - \lambda_{1} ( e_{1 \times n} W_{n \times 1} - 1) \\ \\ \frac{\partial L(W, \lambda_{1})}{\partial W}& = & W_{1 \times n}^{T} \Sigma_{n \times n} - \lambda_{1} e_{1 \times n} = 0\\ \frac{\partial L(W, \lambda_{1})}{\partial \lambda_1} & = & - e_{1 \times n} W_{n \times 1} + 1 = 0\\ \end{eqnarray*} \]

Matrix Form

Minimal Variance (No Risk Free rate and no target return)

\[ \begin{equation*} \begin{bmatrix} \Sigma && - e \\ e^{T} && 0 \\ \end{bmatrix} \begin{bmatrix} x \\ \lambda_{1} \\ \end{bmatrix}=\begin{bmatrix} 0 \\ 1 \\ \end{bmatrix} \end{equation*} \]

def min_var():
    A = np.zeros((n + 1, n + 1))
    A[0:n, 0:n] = covariance
    A[n, 0:n] = + np.ones(n)
    A[0:n, n] = - np.ones(n)

    B = np.zeros((n + 1, 1))
    B[n] = 1

    # we save the weights
    return blas.solve(A, B)[0:n, 0]

variance_min = min_var()

std_min = np.sqrt(portfolio_variance(variance_min)) * 100
ret_min = portfolio_return(variance_min)

Efficient Frontier (No Risk Free rate and target return)

\[ \begin{equation*} \begin{bmatrix} \Sigma && - e && - R \\ e^{T} && 0 && 0 \\ R^{T} && 0 && 0 \end{bmatrix} \begin{bmatrix} x \\ \lambda_{1} \\ \lambda_{2} \\ \end{bmatrix}=\begin{bmatrix} 0 \\ 1 \\ r_{t} \end{bmatrix} \end{equation*} \]

def efficient_frontier():
    A = np.zeros((n + 2, n + 2))
    A[0:n, 0:n] = covariance

    A[n, 0:n] = + np.ones(n)
    A[0:n, n] = - np.ones(n)

    A[n + 1, 0:n] = + expected_returns.values
    A[0:n, n + 1] = - expected_returns.values

    B = np.zeros((n + 2, m))
    B[n, 0:m] = np.ones(m)
    B[n + 1, 0:m] = target_returns

    return blas.solve(A, B)[0:n, 0:m]

def capital_allocation_line():
    m = len(target_returns)

    A = np.zeros((n + 3, n + 3))
    A[0:n, 0:n] = covariance

    A[n + 1, 0:n + 1] = + np.ones(n + 1)
    A[0:n + 1, n + 1] = - np.ones( + 1)

    A[n + 2, 0:n] = + expected_returns.values
    A[0:n, n + 2] = - expected_returns.values

    A[n + 2, n] = + risk_free_rate
    A[n, n + 2] = - risk_free_rate

    B = np.zeros((n + 3, m))
    B[n + 1, 0:m] = np.ones(m)
    B[n + 2, 0:m] = target_returns

    return blas.solve(A, B)[0:n + 1, 0:m]

cal = capital_allocation_line()

Optimal Portfolio

The optimal portfolio is the portfolio which is on the Capital Allocation line and on the Efficient frontier. the capital Allocation line is defined such as $ r = a + b $. To find the optimal portfolio, we only need to find \(w\) such as : \[ w^T R = a \sqrt{w^T \Sigma w} + b \]

def std(w):
    return np.sqrt(w.dot(covariance).dot(np.transpose(w)))

def optimal_portfolio():
    weight = variance_min
    weight = blas.solve(covariance, expected_returns * (std(weight) / a))
    return weight / sum(weight)    

optimal = np.zeros(n + 1)
